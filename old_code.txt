    async def get_available_formats(self, url: str):
        """Get available video formats with retry mechanism"""
        formats, _ = await self.get_formats_with_retry(url)
        return formats

    async def _get_formats_internal(self, url: str):
        """Internal method for getting formats without retry"""
        def _get_formats():
            with yt_dlp.YoutubeDL(self.base_ydl_opts) as ydl:
                info = ydl.extract_info(url, download=False)
                formats = info.get("formats", [])
                seen = set()
                filtered = []

                valid_resolutions = {"360p", "480p", "720p", "1080p", "1440p", "2160p"}
                audio_formats = [f for f in formats if f.get("acodec") != "none" and f.get("vcodec") == "none"]
                best_audio = max(audio_formats, key=lambda x: x.get("abr") or 0) if audio_formats else None
                audio_size = best_audio.get("filesize") or best_audio.get("filesize_approx") or 0 if best_audio else 0

                for f in formats:
                    if f.get("vcodec") == "none":
                        continue

                    res = f.get("format_note") or f.get("resolution") or f.get("height")
                    if isinstance(res, int):
                        res = f"{res}p"
                    if not res or res not in valid_resolutions or res in seen:
                        continue

                    seen.add(res)
                    video_size = f.get("filesize") or f.get("filesize_approx") or 0
                    total_size = video_size + audio_size if video_size and audio_size else video_size or 0

                    # Format size for display
                    if total_size < 1024 * 1024:  # Less than 1MB
                        size_str = f"{round(total_size / 1024, 2)} KB"
                    else:
                        size_str = f"{round(total_size / (1024 * 1024), 2)} MB"

                    filtered.append({
                        "format_id": f["format_id"],
                        "resolution": res,
                        "filesize": total_size,
                        "size_str": size_str
                    })

                filtered.sort(key=lambda x: int(x["resolution"].rstrip("p")))
                return filtered

        return await self._run_in_executor(_get_formats)

    async def get_formats_with_retry(self, url: str, max_retries: int = 3) -> tuple[list, bool]:
        """Get video formats with retries and auto-update"""
        formats = []
        updated = False

        for attempt in range(max_retries):
            try:
                formats = await self._get_formats_internal(url)
                if formats:
                    return formats, updated

                if attempt == 0:
                    updated = await self.update_ytdlp()
                    if updated:
                        self.logger.info("yt-dlp updated successfully, retrying...")
                    await asyncio.sleep(1)
                else:
                    await asyncio.sleep(1)

            except Exception as e:
                self.logger.warning(f"Format fetch attempt {attempt + 1} failed: {e}")
                if attempt == max_retries - 1:
                    raise

        return formats, updated

    async def download_video_by_format(self, url: str, format_id: str, progress_callback=None):
        """Download video in specific format with audio"""
        async with self.semaphore:
            def _download():
                ydl_opts = {
                    **self.base_ydl_opts,
                    'format': f"{format_id}+bestaudio",
                    'progress_hooks': [progress_callback] if progress_callback else [],
                    'merge_output_format': 'mp4',
                    'postprocessors': [{
                        'key': 'FFmpegVideoConvertor',
                        'preferedformat': 'mp4'
                    }]
                }

                with yt_dlp.YoutubeDL(ydl_opts) as ydl:
                    try:
                        info = ydl.extract_info(url, download=True)
                        if not info:
                            self.logger.error(f"Failed to download video: {url}")
                            return None

                        filename = ydl.prepare_filename(info)
                        if not os.path.exists(filename):
                            self.logger.error(f"File not found: {filename}")
                            return None

                        return filename
                    except Exception as e:
                        self.logger.error(f"Download error: {e}")
                        return None

            return await self._run_in_executor(_download)

    async def update_ytdlp(self):
        """Update yt-dlp to the latest version"""
        try:
            self.logger.info("Updating yt-dlp...")
            process = await asyncio.create_subprocess_exec(
                'yt-dlp', '-U',
                stdout=asyncio.subprocess.PIPE,
                stderr=asyncio.subprocess.PIPE
            )
            await process.communicate()
            return True
        except Exception as e:
            self.logger.error(f"Failed to update yt-dlp: {e}")
            return False

    async def cleanup(self):
        """Clean up temporary files"""

            await asyncio.to_thread(shutil.rmtree, self.download_dir, ignore_errors=True)
            self.logger.info("Temporary files cleaned up")

        patterns = [

            r'https?://(www\.)?vk\.ru/(video|audio|clip|wall).+',
            r'https?://(www\.)?m\.vk\.com/(video|audio|clip|wall).+',
        ]
        return any(bool(re.match(pattern, url)) for pattern in patterns)

    async def download(self, urls: Union[str, List[str]], remove_watermark: bool = False) -> Optional[Dict[str, Any]]:
        if isinstance(urls, str):
            urls = [url.strip() for url in urls.split(',') if url.strip()]

        valid_urls = [url for url in urls if self.is_valid_url(url)]
        if not valid_urls:
            self.logger.error("❌ No valid VK URLs provided")
            return None

        loop = asyncio.get_event_loop()
        tasks = [loop.run_in_executor(None, self._download_video, url, remove_watermark) for url in valid_urls]
        results = await asyncio.gather(*tasks)

        for res in results:
            if res:  # Если хоть один файл скачался — возвращаем

        return None

                if os.path.exists(filepath) and os.path.isfile(filepath) and os.path.getsize(filepath) > 0:

class Pinterest:
    def __init__(self, output_path: str = None):
        self.output_path = output_path or tempfile.mkdtemp(prefix="yt_cache_")
        self.executor = ThreadPoolExecutor(max_workers=16)
        self.semaphore = asyncio.Semaphore(8)
        self.BASE_API_URL = "https://www.pinterest.com/resource/BaseSearchResource/get/"
        self.HEADERS = {
            "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 "
                      "(KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36"
        }
        self.logger_handler = LoggerHandler("Pinterest Downloader")
        self.logger = self.logger_handler.setup_logger_handler()
        self.logger.info(f"Pinterest Downloader Initialized")

    async def __aenter__(self):
        """Асинхронный вход в контекстный менеджер."""
        return self

    async def __aexit__(self, exc_type, exc_val, exc_tb):
        """Асинхронный выход из контекстного менеджера."""
        self.cleanup()

    @asynccontextmanager
    async def aiohttp_session_scope(self) -> AsyncGenerator[aiohttp.ClientSession, None]:
        """Асинхронный контекстный менеджер для aiohttp сессии с обработкой SSL-ошибок."""
        timeout = aiohttp.ClientTimeout(total=30, connect=10, sock_read=10)
        connector = aiohttp.TCPConnector(limit=16,ttl_dns_cache=300,use_dns_cache=True,force_close=False,enable_cleanup_closed=True,ssl=False)
        session = aiohttp.ClientSession(headers=self.HEADERS,timeout=timeout,connector=connector,trust_env=True)

        try:
            yield session
        except aiohttp.ClientError as e:
            self.logger.error(f"Network error: {e}")
            raise
        except Exception as e:
            self.logger.error(f"Unexpected error: {e}")
            raise
        finally:
            try:
                if not session.closed:
                    await session.close()
            except Exception as e:
                self.logger.error(f"Error closing session: {e}")

    async def search(self, query: str) -> List[Dict[str, str]]:
        async with self.aiohttp_session_scope() as session:
            try:
                self.logger.debug(f"Searching for: {query}")
                params = {
                    "source_url": f"/search/pins/?q={query}&rs=typed",
                    "data": json.dumps({"options": {"query": query, "scope": "pins"}})
                }
                async with session.get(self.BASE_API_URL, params=params, headers=self.HEADERS) as response:
                    data = await response.json()
                    results = [
                        {
                            "url": item["images"]["orig"]["url"],
                            "thumbnail": item["images"]["236x"]["url"]
                        }
                        for item in data.get("resource_response", {}).get("data", {}).get("results", [])
                        if "images" in item and "orig" in item["images"] and "236x" in item["images"]
                    ]
                    self.logger.info(f"Found {len(results)} results for query: {query}")
                    return results
            except Exception as e:
                self.logger.error(f"Search failed for query '{query}': {e}")
                return []

    async def download(self, pin_url: str, return_url: bool = False) -> Dict[str, Any]:
        async with self.aiohttp_session_scope() as session:
            try:
                self.logger.debug(f"Starting download for: {pin_url}")
                async with session.head(pin_url, allow_redirects=True) as response:
                    full_url = str(response.url)

                async with session.get(full_url, headers=self.HEADERS) as response:
                    page = await response.text()

                # Try extract video
                video_match = re.search(r'"contentUrl"\s*:\s*"([^"]+\.mp4[^"]*)"', page)
                if video_match:
                    video_url = video_match.group(1).replace('\\u002F', '/').replace('\\', '')
                    self.logger.info(f"Found video: {video_url}")
                    if return_url:
                        return {"success": True, "type": "video", "url": video_url}

                    return await self._save_file(session, video_url, "pinterest_video_")

                # Try extract image
                image_match = re.search(r'https://i\.pinimg\.com/originals/[^"]+', page)
                if image_match:
                    image_url = image_match.group(0)
                    self.logger.info(f"Found image: {image_url}")
                    if return_url:
                        return {"success": True, "type": "image", "url": image_url}

                    return await self._save_file(session, image_url, "pinterest_image_")

                self.logger.warning(f"No media found for {pin_url}")
            except Exception as e:
                self.logger.exception(f"Download failed for {pin_url}: {e}")

    async def _save_file(self, session: aiohttp.ClientSession, media_url: str, prefix: str) -> Dict[str, Any]:
        try:
            self.logger.debug(f"Saving file from: {media_url}")
            async with session.get(media_url, headers=self.HEADERS) as resp:
                data = await resp.read()

            filename = prefix + os.path.basename(media_url)
            filepath = os.path.join(self.output_path, filename)
            async with aiofiles.open(filepath, 'wb') as f:
                await f.write(data)

            self.logger.info(f"Saved {'video' if prefix == 'pinterest_video_' else 'image'} to {filepath}")

            return {
                "success": True,
                "file_path": filepath,
                "title": os.path.basename(filepath),
                "type": "video" if prefix == "pinterest_video_" else "image"
            }
        except Exception as e:
            self.logger.exception(f"Failed to save file from {media_url}: {e}")
            return {"success": False}

    async def cleanup(self):
        """Очистка ресурсов"""
        try:
            if os.path.exists(self.output_path):
                await asyncio.to_thread(shutil.rmtree, self.output_path, ignore_errors=True)
            self.executor.shutdown(wait=False)
            self.logger.info("Resources cleaned up")
        except Exception as e:
            self.logger.error(f"Error during cleanup: {e}", exc_info=True)

import asyncio, instaloader, tempfile, shutil

from handler import logger
from typing import Optional, Dict, List, Union

from urllib.parse import urlparse


class Instagram:
    def __init__(self, url: str):
        self.url = url
        self.output_path = Path(tempfile.mkdtemp(prefix="instagram_cache"))
        self.loader = instaloader.Instaloader(
            download_comments=False,
            download_video_thumbnails=False,
            save_metadata=False,
            download_geotags=False,
            post_metadata_txt_pattern="",
            storyitem_metadata_txt_pattern="",
        )
        self.logger_handler = logger.LoggerHandler("Instagram")
        self.logger = self.logger_handler.setup_logger_handler()
        self.logger.info(f"Instagram Downloader Initialized With URL: {url}")

    async def __aenter__(self):
        return self

    async def __aexit__(self, exc_type, exc_val, exc_tb):
        self.cleanup()
        return self

    async def download(self) -> Union[Dict, List[Dict], None]:
        try:
            content_type = self._detect_content_type(self.url)
            self.logger.info(f"Detected content type: {content_type}")

            match content_type:
                case "post":
                    result = await self._handle_post()
                case "story":
                    result = await self._handle_story()
                case "reel":
                    result = await self._handle_reel()
                case "profile":
                    result = await self._handle_profile()
                case _:
                    raise ValueError(f"Unsupported content type: {content_type}")

            if result is None:
                self.logger.warning(f"No content found for {content_type} at {self.url}")
                return None

            self.logger.info(f"Successfully downloaded {content_type} content")
            return result

        except Exception as e:
            self.logger.error(f"[ ERROR ] Downloading content: {e}", exc_info=True)
            return None

    def _detect_content_type(self, url: str) -> str:
        try:
            if "/p/" in url:
                return "post"
            elif "/stories/" in url:
                return "story"
            elif "/reel/" in url:
                return "reel"
            elif "/" in url and not any(x in url for x in ["/p/", "/stories/", "/reel/"]):
                return "profile"
            else:
                raise ValueError(f"Unsupported URL format: {url}")
        except Exception as e:
            self.logger.error(f"Error detecting content type: {e}")
            raise

    async def _handle_post(self) -> Optional[Dict]:
        try:
            post = await self._get_post(self.url)
            if not post:
                self.logger.warning(f"Post not found for URL: {self.url}")
                return None

            files = await self._download_post(post)
            if not files:
                self.logger.warning(f"No files downloaded for post: {post.shortcode}")
                return None

            return {
                "type": "post",
                "files": files,
                "caption": post.caption or "Instagram post",
                "shortcode": post.shortcode,
                "link": f"https://www.instagram.com/p/{post.shortcode}/"
            }
        except Exception as e:
            self.logger.error(f"Error handling post: {e}", exc_info=True)
            return None

    async def _handle_reel(self) -> Optional[Dict]:
        try:
            post = await self._get_post(self.url)
            if not post:
                self.logger.warning(f"Reel not found for URL: {self.url}")
                return None

            if not post.is_video:
                self.logger.warning(f"Post is not a video: {post.shortcode}")
                return None

            files = await self._download_post(post)
            if not files:
                self.logger.warning(f"No files downloaded for reel: {post.shortcode}")
                return None

            return {
                "type": "reel",
                "files": files,
                "caption": post.caption or "Instagram reel",
                "shortcode": post.shortcode,
                "link": f"https://www.instagram.com/reel/{post.shortcode}/"
            }
        except Exception as e:
            self.logger.error(f"Error handling reel: {e}", exc_info=True)
            return None

    async def _handle_story(self) -> Optional[List[Dict]]:
        try:
            username = self.url.split("/stories/")[1].split("/")[0]
            self.logger.info(f"Processing story for user: {username}")

            profile = await asyncio.to_thread(Profile.from_username, self.loader.context, username)
            story_items = await asyncio.to_thread(lambda: list(self.loader.get_stories(userids=[profile.userid])))

            if not story_items:
                self.logger.warning(f"No stories found for user: {username}")
                return None

            all_stories = []
            for story in story_items:
                for item in story.get_items():
                    file = await self._download_story(item)
                    if file:
                        all_stories.append({
                            "type": "story",
                            "file": file,
                            "date": item.date,
                            "link": f"https://www.instagram.com/stories/{username}/{item.mediaid}/"
                        })

            if not all_stories:
                self.logger.warning(f"No stories downloaded for user: {username}")
                return None

            return all_stories
        except Exception as e:
            self.logger.error(f"Error handling story: {e}", exc_info=True)
            return None

    async def _handle_profile(self) -> Optional[List[Dict]]:
        try:
            username = self.url.rstrip("/").split("/")[-1]
            self.logger.info(f"Processing profile: {username}")

            profile = await asyncio.to_thread(Profile.from_username, self.loader.context, username)
            downloaded_posts = []

            for post in profile.get_posts():
                files = await self._download_post(post)
                if files:
                    downloaded_posts.append({
                        "type": "post",
                        "files": files,
                        "caption": post.caption or "Instagram post",
                        "shortcode": post.shortcode,
                        "link": f"https://www.instagram.com/p/{post.shortcode}/"
                    })
                    await self.cleanup()
                    self.output_path = Path(tempfile.mkdtemp(prefix="instagram_cache"))

            if not downloaded_posts:
                self.logger.warning(f"No posts downloaded for profile: {username}")
                return None

            return downloaded_posts
        except Exception as e:
            self.logger.error(f"Error handling profile: {e}", exc_info=True)
            return None

    async def _get_post(self, url: str) -> Optional[Post]:
        try:
            path = urlparse(url).path
            shortcode = path.rstrip("/").split("/")[-1]
            self.logger.info(f"Extracting post with shortcode: {shortcode}")
            return await asyncio.to_thread(Post.from_shortcode, self.loader.context, shortcode)
        except Exception as e:
            self.logger.error(f"Error getting post: {e}", exc_info=True)
            return None

    async def _download_post(self, post: Post) -> Optional[List[str]]:
        try:
            target_path = self.output_path / post.shortcode
            target_path.mkdir(parents=True, exist_ok=True)

            await asyncio.to_thread(self.loader.download_post, post, target_path)

            # Удаление метаданных
            for ext in (".json.xz", ".txt"):
                for f in target_path.glob(f"*{ext}"):
                    f.unlink()

            files = [str(f) for f in target_path.glob("*") if f.suffix in (".jpg", ".jpeg", ".png", ".mp4")]
            if not files:
                self.logger.warning(f"No media files found for post: {post.shortcode}")
                return None

            return files
        except Exception as e:
            self.logger.error(f"Error downloading post: {e}", exc_info=True)
            return None

    async def _download_story(self, story_item: StoryItem) -> Optional[str]:
        try:
            target_path = self.output_path / "stories"
            target_path.mkdir(parents=True, exist_ok=True)
            await asyncio.to_thread(self.loader.download_storyitem, story_item, target=target_path)

            # Удаление метаданных
            for f in target_path.glob("*.txt"):
                f.unlink()
            for f in target_path.glob("*.json.xz"):
                f.unlink()

            media_file = next((str(f) for f in target_path.iterdir() if f.suffix in (".jpg", ".jpeg", ".png", ".mp4")), None)
            if not media_file:
                self.logger.warning(f"No media file found for story item: {story_item.mediaid}")
                return None

            return media_file
        except Exception as e:
            self.logger.error(f"Error downloading story: {e}", exc_info=True)
            return None

    async def cleanup(self):
        try:
            await asyncio.to_thread(shutil.rmtree, self.output_path, ignore_errors=True)
            self.logger.info("Temporary files cleaned up")
        except Exception as e:
            self.logger.error(f"Error cleaning up: {e}", exc_info=True)
